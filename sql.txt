CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(150),
    email VARCHAR(150) UNIQUE,
    phone VARCHAR(20),
    password VARCHAR(255),
    role ENUM('admin','member','staff') DEFAULT 'member',
    status ENUM('active','blocked') DEFAULT 'active',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE user_profiles (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT,
    address TEXT,
    city VARCHAR(100),
    state VARCHAR(100),
    pincode VARCHAR(10),
    dob DATE,
    gender VARCHAR(10),
    profile_image VARCHAR(255),
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);

CREATE TABLE designations (
    id INT AUTO_INCREMENT PRIMARY KEY,
    title VARCHAR(100),
    fee DECIMAL(10,2),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE memberships (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT,
    designation_id INT,
    join_date DATE,
    expiry_date DATE,
    status ENUM('active','expired','blocked') DEFAULT 'active',
    referral_code VARCHAR(50),
    referred_by INT NULL,
    FOREIGN KEY (user_id) REFERENCES users(id),
    FOREIGN KEY (designation_id) REFERENCES designations(id),
    FOREIGN KEY (referred_by) REFERENCES users(id)
);

CREATE TABLE transactions (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT,
    type ENUM('membership','donation','event','custom'),
    reference_id INT,
    amount DECIMAL(10,2),
    payment_method ENUM('razorpay','payu','phonepe','cash','bank'),
    status ENUM('pending','success','failed'),
    transaction_id VARCHAR(150),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id)
);

CREATE TABLE donations (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT,
    campaign_id INT NULL,
    amount DECIMAL(10,2),
    donation_type ENUM('online','cash','custom'),
    receipt_no VARCHAR(100),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id)
);

CREATE TABLE campaigns (
    id INT AUTO_INCREMENT PRIMARY KEY,
    title VARCHAR(200),
    description TEXT,
    goal_amount DECIMAL(12,2),
    start_date DATE,
    end_date DATE,
    status ENUM('active','completed','closed') DEFAULT 'active',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE receipts (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT,
    type ENUM('membership','donation','event','custom'),
    reference_id INT,
    receipt_no VARCHAR(100),
    qr_code VARCHAR(255),
    pdf_path VARCHAR(255),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE certificates (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NULL,
    name VARCHAR(150),
    type ENUM('membership','achievement','internship','visitor'),
    template_id INT,
    qr_code VARCHAR(255),
    pdf_path VARCHAR(255),
    issued_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE certificate_templates (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100),
    file_path VARCHAR(255)
);

CREATE TABLE id_cards (
    id INT AUTO_INCREMENT PRIMARY KEY,

    issued_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE appointment_letters (
    id INT AUTO_INCREMENT PRIMARY KEY,

    pdf_path VARCHAR(255),

);

CREATE TABLE news (
